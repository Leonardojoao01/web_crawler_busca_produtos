        function atualizarMetaData(idFreteEntregaTipo) {
            var shippingPrice;
            var shippingMaxDays;
            var zipCode;

            if ('undefined' == typeof (window._siteMetadataChange)) {
                window._siteMetadataChange = [];
            }

            var cep1 = $('#txtCep1').val();
            var cep2 = $('#txtCep2').val();
            var precoFrete = $('[data-id=endereco_entrega' + idFreteEntregaTipo + '_valor]').html();
            var diasFrete = $('[data-id=endereco_entrega' + idFreteEntregaTipo + '_tempo]').html();

            if (typeof (precoFrete) != 'undefined')
                shippingPrice = (precoFrete.replace(/[^0-9,]/g, '')).replace(',', '.');

            if (typeof (diasFrete) != 'undefined')
                shippingMaxDays = diasFrete.replace(/[^0-9,]/g, '');

            if ((typeof (cep1) != '') && ((cep2) != ''))
                zipCode = cep1.concat(cep2);

            if (!$.isNumeric(shippingPrice)) {
                if (precoFrete.toLowerCase().replace('á', 'a') == 'gratis') {
                    shippingPrice = 0;
                }
            }

            var detail = { page: { product: {} } };

            if (typeof (shippingPrice) != 'undefined') {
                window.siteMetadata.page.product['shippingPrice'] = shippingPrice;
                detail.page.product['shippingPrice'] = shippingPrice;
            }

            if (typeof (shippingMaxDays) != 'undefined') {
                window.siteMetadata.page.product['shippingMaxDays'] = shippingMaxDays;
                detail.page.product['shippingMaxDays'] = shippingMaxDays;
            }

            if (typeof (shippingMaxDays) != 'zipCode') {
                window.siteMetadata.page.product['zipCode'] = zipCode;
                detail.page.product['zipCode'] = zipCode;
            }

            window._siteMetadataChange.push(['product.shipping', detail]);
        }

        function listarEntregas(data) {
            var entregas = $('[data-id=endereco_entrega]');

            var item, entrega;

            desativarEntregas();
            entregas.show();

            var dataChegada = $('[data-id=data_chegada]').html();

                item = data.ClienteEnderecoFreteEntregasTipos[i];
                entrega = entregas.find('[data-id=endereco_entrega' + item.IdFreteEntregaTipo + '_valor]');

                if (entrega.length > 0) {
                    entrega.text(item.ValorFreteComDesconto != 0 ? FormatarValor(item.ValorFreteComDesconto) : "Grátis");

                    if (item.IdFreteEntregaTipo != 12) {
                        if (item.PrazoEntrega == 0)
                            $('[data-id=endereco_entrega' + item.IdFreteEntregaTipo + '_tempo]').text("HOJE");
                        else
                            $('[data-id=endereco_entrega' + item.IdFreteEntregaTipo + '_tempo]').text(item.PrazoEntrega + " dia" + (item.PrazoEntrega > 1 ? "s" : ""));
                    }
                    else {
                        $('[data-id=endereco_entrega' + item.IdFreteEntregaTipo + '_tempo]').text("A partir de " + item.PrazoEntrega + " dia" + (item.PrazoEntrega > 1 ? "s" : ""));
                        if (typeof (dataChegada) != 'undefined')
                            $('[data-id=endereco_entrega' + item.IdFreteEntregaTipo + '_tempo]').text("A partir de " + dataChegada)
                    }

                    entrega = entrega.parent();
                    $('[data-id=endereco_entrega' + item.IdFreteEntregaTipo + ']').prop('disabled', false).parents("[data-id=grp]").removeClass("off");
                }
            }
        }

        function FormatarValor(valor) {
            var strValor = valor.toString();
            if (strValor.indexOf(".") < 0)
                strValor += ".00";
            else if (strValor.substring(strValor.indexOf(".") + 1).length == 1)
                strValor += "0";
            strValor = strValor.replace(".", ",");
            if (valor > 1000) {
                var arr = strValor.split(",");
                strValor = arr[0].substring(0, arr[0].length - 3) + "." + arr[0].substring(arr[0].length - 3) + "," + arr[1];
            }
            strValor = "R$ " + strValor;
            return strValor;
        }
